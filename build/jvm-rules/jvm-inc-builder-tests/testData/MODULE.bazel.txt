# hermetic_cc_toolchain (zig) is used in rules_jvm but marked as dev_dependency there.
# Do NOT register zig toolchains here: the zig C++ toolchain uses x86_64-windows-gnu
# which lacks pthread.h, causing protobuf/abseil-cpp compilation to fail on Windows.
# Without registration, Bazel falls back to the auto-detected system C++ toolchain (MSVC).
bazel_dep(name = "hermetic_cc_toolchain", version = "4.0.1")
toolchains = use_extension("@hermetic_cc_toolchain//toolchain:ext.bzl", "toolchains")
use_repo(toolchains, "zig_sdk")
