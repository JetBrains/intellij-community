Reports redundant with function calls that don't access anything from the receiver.

Examples:


  class MyClass {
      fun f(): String = ""
  }

  fun testRedundant() {
      with(c) { // <== 'with' is redundant since 'c' isn't used
          println("1")
      }
  }

  fun testOk() {
      val c = MyClass()
      with(c) { // <== OK because 'f()' is effectively 'c.f()'
          println(f())
      }
  }
