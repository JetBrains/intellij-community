Reports when a method decorated with @override doesn't have a matching method in its ancestor classes

Example:


from typing import override

class Parent:
    def foo(self) -> int:
        return 1

    def bar(self, x: str) -> str:
        return x

class Child(Parent):
    @override
    def foo(self) -> int:
        return 2

    @override # Missing super method for override function
    def baz(self) -> int:
        return 1
